class Solution:
    def levelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:
        queue = deque()
        if root != None:
            queue.append(root)
        result = []
        while queue:
            level = []
            size = len(queue)
            for I in range(size):
                node = queue.popleft()
                level.append(node.val)
                if node.left != None:
                    queue.append(node.left)
                if node.right != None:
                    queue.append(node.right)
            result.append(level)
        return result

        if root == None:
            return False 
        targetSum = targetSum - root.val
        if root.left == None and root.right == None:
            return targetSum == 0
        return self.hasPathSum(root.left, targetSum) or self.hasPathSum(root.right, targetSum)

class Solution:
    def dfs(self, node, target, path, result): 
        if node == None:
            return 
        path.append(node.val)
        target = target - node.val 
        if node.left == None and node.right == None and target == 0:
            result.append(list(path))
        else:
            self.dfs(node.left, target, path, result)
            self.dfs(node.right, target, path, result)        
        path.pop()
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:
        result = []
        path = []
        self.dfs(root, targetSum, path, result)
        return result

class Solution:
    def doSum(self, node, current):
        if node == None:
            return 0
        current = current * 10 + node.val 
        if node.left == None and node.right == None:
            return current 
        leftSum = self.doSum(node.left, current)
        rightSum = self.doSum(node.right, current)
        return leftSum + rightSum
