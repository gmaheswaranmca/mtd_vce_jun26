class Solution:
    def hasCycle(self, head: Optional[ListNode]) -> bool:
        slow = head
        fast = head 
        while fast != None and fast.next != None:
            slow = slow.next 
            fast = fast.next.next
            if slow == fast:
                return True 
        return False

#include <iostream>
using namespace std;
int fact(int n) { 
    if(n <= 1) {
        return 1;
    }
    return n * fact(n - 1);
}
int ifact(int n) { 
    int r = 1;
    for(int i = n; i >= 1; i--) {
        r *= i;
    }
    return r;
}
int main() {
    int n;
    cin >> n;
    cout << "Factorial (Non-Recursive): " << ifact(n) << endl;
    cout << "Factorial (Recursive): " << fact(n);
    return 0;
}

def fibonocci(n):
    if n < 0 :
        return -1
    if n == 1: 
        return 0
    if n == 2:
        return 1
    return fibonocci(n-1) + fibonocci(n-2)

class Solution:
    def backtrack(self, nums: List[int], index:int, n:int,
        current:List[int], answer:List[List]) :
        if index == n:
            answer.append(list(current))
            return 
        current.append(nums[index])
        self.backtrack(nums, index + 1, n, current, answer)
        current.pop()
        self.backtrack(nums, index+1, n, current, answer)
    def subsets(self, nums: List[int]) -> List[List[int]]:
        current = []
        answer = []
        self.backtrack(nums, 0, len(nums), current, answer)
        return answer

import java.util.ArrayList;
import java.util.List;

class Solution {
    private void backtrack(int[] nums, int index, int n, 
                           List<Integer> current, List<List<Integer>> answer) {
        if (index == n) {
            answer.add(new ArrayList<>(current));
            return;
        }
        current.add(nums[index]);
        backtrack(nums, index + 1, n, current, answer);
        current.remove(current.size() - 1);
        backtrack(nums, index + 1, n, current, answer);
    }

    public List<List<Integer>> subsets(int[] nums) {
        List<Integer> current = new ArrayList<>();
        List<List<Integer>> answer = new ArrayList<>();        
        backtrack(nums, 0, nums.length, current, answer);        
        return answer;
    }
}
