Pseudocode - insertion sort 
subroutine doSort(nums, n) 
    for i <- 1 .. n-1
        out = nums[i]
        j = i - 1
        while j >= 0 && nums[j] > out
            nums[j+1] = nums[j]
            j --
        if i != (j + 1)
            nums[j + 1] = out

cpp imple:

#include<iostream>
#include<vector>
using namespace std;
void doSort(vector<int>& nums, int n) {
    for(int I = 1; I < n; I++) {
        int taken = nums[I];
        int J = I - 1;
        while( J >= 0 && nums[J] > taken) {
            nums[J+1] = nums[J];
            J--;
        }
        if( I != (J + 1)) {
            nums[J + 1] = taken;
        }
    }
}

int main() {
    int n;
    vector<int> nums;
    cin >> n;
    for(int I = 0; I < n; I++) {
        int num;
        cin >> num;
        nums.push_back(num);
    }
    doSort(nums, n);
    for(int I = 0; I < n; I++) {
        if(I == (n-1)) {
            cout << nums[I];
        } else {
            cout << nums[I] << " ";
        }
    }
    return 0;
}

Quick Sort: 

Algorithm:
Choose a pivot element
Partition array so elements < pivot are on left, > pivot on right
Recursively apply quicksort to left and right subarrays
Combine results

---
pseudo:
FUNCTION Partition(arr, low, high)
    pivot ← arr[high]
    i ← low - 1

    FOR j ← low TO high - 1
        IF arr[j] ≤ pivot THEN
            i ← i + 1

            temp ← arr[i]
            arr[i] ← arr[j]
            arr[j] ← temp
        END IF
    END FOR
    IF (i + 1) != high THEN
        temp ← arr[i + 1]
        arr[i + 1] ← arr[high]
        arr[high] ← temp
    END IF
    RETURN i + 1
END FUNCTION
FUNCTION QuickSort(arr, low, high)
    IF low < high THEN
        pivotIndex ← Partition(arr, low, high)
        QuickSort(arr, low, pivotIndex - 1)
        QuickSort(arr, pivotIndex + 1, high)
    END IF
END FUNCTION





Time Complexity:
    Best Case: O(n log n) - Pivot divides array into equal halves
    Average Case: O(n log n)
    Worst Case: O(n²) - Pivot is always smallest/largest element
Space Complexity: O(log n) - Due to recursion stack
Characteristics:
    Stable: No (depends on partition implementation)
    In-place: Yes
    Adaptive: No
Advantages:
    Very efficient on average O(n log n)
    In-place sorting
    Widely used in practice
    Cache-friendly
Disadvantages:
    Worst-case O(n²) time complexity
    Not stable
    Performance depends on pivot selection

Merge Sort is a divide-and-conquer algorithm that divides the array into halves, sorts them, and then merges them back.

Algorithm:
    Divide array into two halves
    Recursively sort both halves
    Merge the sorted halves
Time Complexity:
    Best Case: O(n log n)
    Average Case: O(n log n)
    Worst Case: O(n log n) - Consistent performance
Space Complexity: O(n) - Requires additional space for temporary arrays
Characteristics:
    Stable: Yes
    In-place: No
    Adaptive: No
Advantages:
    Guaranteed O(n log n) time complexity
    Stable sorting algorithm
    Predictable performance
    Good for large datasets
    Parallelizable
Disadvantages:
    Requires O(n) extra space
    Not in-place
Slower than quicksort in practice for smaller arrays

FUNCTION Merge(arr, left, mid, right)
    CREATE temp[]

    i ← left
    j ← mid + 1

    WHILE i ≤ mid AND j ≤ right
        IF arr[i] ≤ arr[j] THEN
            APPEND arr[i] TO temp
            i ← i + 1
        ELSE
            APPEND arr[j] TO temp
            j ← j + 1
        END IF
    END WHILE

    WHILE i ≤ mid
        APPEND arr[i] TO temp
        i ← i + 1
    END WHILE

    WHILE j ≤ right
        APPEND arr[j] TO temp
        j ← j + 1
    END WHILE

    COPY temp BACK TO arr[left...right]
END FUNCTION

FUNCTION MergeSort(arr, left, right)
    IF left < right THEN
        mid ← (left + right) / 2

        MergeSort(arr, left, mid)
        MergeSort(arr, mid + 1, right)

        Merge(arr, left, mid, right)
    END IF
END FUNCTION

## C++

```cpp
#include <iostream>
#include <vector>
using namespace std;

void merge(vector<int>& arr, int left, int mid, int right)
{
    vector<int> temp;

    int i = left;
    int j = mid + 1;

    while (i <= mid && j <= right)
    {
        if (arr[i] <= arr[j])
        {
            temp.push_back(arr[i]);
            i++;
        }
        else
        {
            temp.push_back(arr[j]);
            j++;
        }
    }

    while (i <= mid)
    {
        temp.push_back(arr[i]);
        i++;
    }

    while (j <= right)
    {
        temp.push_back(arr[j]);
        j++;
    }

    for (int k = 0; k < temp.size(); k++)
    {
        arr[left + k] = temp[k];
    }
}

void mergeSort(vector<int>& arr, int left, int right)
{
    if (left < right)
    {
        int mid = left + (right - left) / 2;

        mergeSort(arr, left, mid);
        mergeSort(arr, mid + 1, right);

        merge(arr, left, mid, right);
    }
}

int main()
{
    int n;
    cin >> n;

    vector<int> arr(n);

    for (int i = 0; i < n; i++)
    {
        cin >> arr[i];
    }

    mergeSort(arr, 0, n - 1);

    for (int i = 0; i < n; i++)
    {
        cout << arr[i];

        if (i < n - 1)
            cout << " ";
    }

    return 0;
}
```

```python
        merge_sort(arr, mid + 1, right)
```
