class Solution:
    def bt(self, nums, n, used, path, res):
        if len(path) == n:
            pathCopy = list(path)
            res.append(pathCopy)
            return 
        for I in range(n):
            if used[I]:
                continue 
            path.append(nums[I])
            used[I] = True 
            self.bt(nums, n, used, path, res)
            path.pop()
            used[I] = False 
    def permute(self, nums: List[int]) -> List[List[int]]:
        res = [] # [[]]
        path = [] 
        n = len(nums)
        used = [False for e in range(n)]
        self.bt(nums, n, used, path, res)
        return res

n-queen 
FUNCTION hasConflict(row, col, cols, diag1, diag2)
    IF (col IN cols) OR
       (row - col) IN diag1 OR
       (row + col) IN diag2
        RETURN TRUE
    RETURN FALSE
FUNCTION updateRowCol(row, col, cols, diag1, diag2)
    ADD col TO cols
    ADD (row - col) TO diag1
    ADD (row + col) TO diag2
FUNCTION removeRowCol(row, col, cols, diag1, diag2)
    REMOVE col FROM cols
    REMOVE (row - col) FROM diag1
    REMOVE (row + col) FROM diag2
FUNCTION backtrack(row, n, board, cols, diag1, diag2)
    IF row = n THEN
        RETURN true 
    FOR col ← 0 TO n-1
        IF hasConflict(row, col, cols, diag1, diag2) THEN
            CONTINUE
        updateRowCol(row, col, cols, diag1, diag2)
        board[row][col] ← 'Q'
        result = backtrack(row + 1, n, board, cols, diag1, diag2)
        if result:
            return true 
        board[row][col] ← '.'
        removeRowCol(row, col, cols, diag1, diag2)
    return result 
FUNCTION solveNQueens(n)
    board ← n × n board filled with '.'
    cols ← empty set
    diag1 ← empty set
    diag2 ← empty set
    result = backtrack(0, n, board, cols, diag1, diag2)
    RETURN result
END FUNCTION
